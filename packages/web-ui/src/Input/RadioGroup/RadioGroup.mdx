# RadioGroup

### Basic usage

<Playground>
  {() => {
    const [value, setValue] = useState();
    return (
      <>
        <label>性别：</label>
        <RadioGroup
          value={value}
          onChange={(value) => setValue(value)}
          options={[
            { value: 1, name: '男' },
            { value: 2, name: '女' },
          ]}
        />
      </>
    );
  }}
</Playground>
