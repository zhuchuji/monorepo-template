### Loading
<Playground>
  {() => {
    const [loading, setLoading] = useState(false);
    return (
      <Select
        loading={loading}
        options={[
          { value: 1, name: 'option 1' },
          { value: 2, name: 'option 2' },
          { value: 3, name: 'option 3' },
        ]}
        onToggleDropdown={(collapsed) => {
          if (collapsed) {
            setLoading(true);
            window.setTimeout(() => {
              setLoading(false);
            }, 4000);
          }
        }}
      />
    );
  }}
</Playground>
