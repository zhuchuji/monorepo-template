import { Playground, Props } from 'docz';
import styled from 'styled-components';
import { Breadcrumb } from '../';

### Basic usage

<Playground>
  {() => {
    const Item = Breadcrumb.Item;
    return (
      <Breadcrumb>
        <Item>Home</Item>
        <Item>Course</Item>
        <Item>Lesson</Item>
      </Breadcrumb>
    );
  }}
</Playground>

### Customization

<Playground>
  {() => {
    const Item = Breadcrumb.Item;
    const BreadcrumbItem = styled(Item)`
      font-size: 30px;
      &.active {
        color: red;
      }
    `;
    const Separator = styled.span`
      margin: 0 10px;
      font-size: 26px;
    `;
    return (
      <Breadcrumb separator={<Separator>/</Separator>}>
        <BreadcrumbItem>Home</BreadcrumbItem>
        <BreadcrumbItem>Course</BreadcrumbItem>
        <BreadcrumbItem>Lesson</BreadcrumbItem>
      </Breadcrumb>
    );
  }}
</Playground>
