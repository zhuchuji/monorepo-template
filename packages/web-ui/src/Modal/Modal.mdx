import { Playground, Props } from 'docz';
import { useState } from 'react';
import { Modal, AlertModal } from '../';

# Modal

### Basic usage

<Playground>
  {() => {
    const [show, toggleShow] = useState(false);
    return (
      <>
        <button onClick={() => toggleShow(true)}>Open Modal</button>
        <Modal title="Basic Modal" open={show} onClose={() => toggleShow(false)}>
          This is a basic modal
        </Modal>
      </>
    );
  }}
</Playground>
